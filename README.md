This program takes a character input from the user (A-Z) and displays its representation using a 2D array of asterisks `*`.

void print_letter(const int pattern[5][5], char symbol) {
    for (int i = 0; i < 5; i++) {
        for (int j = 0; j < 5; j++) {
            printf("%c", pattern[i][j] ? symbol : ' ');
        }
        printf("\n");
    }
}
